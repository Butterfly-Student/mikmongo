LangGraph Agent — Core Logic
Setelah stream dan Go client jalan, bangun agent graph. Mulai dari tools paling dasar: get_customer dan get_billing.
from langgraph.graph import StateGraph, MessagesState, END from langchain_anthropic import ChatAnthropic from langchain_core.tools import tool llm = ChatAnthropic(model="claude-sonnet-4-20250514") @tool async def get_customer_info(name: str) -> dict: """Ambil informasi customer berdasarkan nama.""" return await go_get(f"/api/customers?name={name}") @tool async def get_pppoe_sessions(router_id: str = "") -> dict: """Lihat semua PPPoE session aktif.""" return await go_get("/api/mikrotik/sessions") tools = [get_customer_info, get_pppoe_sessions] llm_with_tools = llm.bind_tools(tools) # Graph sederhana untuk mulai def agent_node(state: MessagesState): response = llm_with_tools.invoke(state["messages"]) return {"messages": [response]}

Intent Router — Hemat LLM Cost
Pertanyaan customer yang sederhana (cek tagihan, status) jangan ke LLM. Rule-based dulu, LLM hanya untuk yang kompleks. Estimasi: hanya ~20% hit LLM.
PATTERNS = { "tagihan": ["tagihan", "bayar", "invoice", "nunggak", "berapa"], "status": ["mati", "lambat", "gangguan", "tidak bisa"], "isolir": ["terisolir", "diblokir", "putus"], } def classify_intent(msg: str) -> str: m = msg.lower() for intent, kws in PATTERNS.items(): if any(k in m for k in kws): return intent return "llm" # fallback ke LLM hanya jika tidak match async def handle_customer(msg: str, customer_id: str): intent = classify_intent(msg) if intent == "tagihan": data = await go_get(f"/api/billing/{customer_id}/latest") return format_billing(data) # tidak ke LLM! return await run_agent(msg, customer_id, mode="customer")
