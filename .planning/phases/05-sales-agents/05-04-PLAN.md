import { useMutation, useQueryClient } from '@tanstack/react-query'
import { toast } from 'sonner'
import { createHotspotSale } from '@/api/hotspot-sale'
import type { CreateHotspotSaleRequest } from '@/lib/schemas/sales'

export function useCreateHotspotSale() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: CreateHotspotSaleRequest) => createHotspotSale(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['hotspot-sales'] })
      toast.success('Penjualan hotspot berhasil dibuat')
    },
    onError: (err: unknown) => {
      const message = (err as { response?: { data?: { error?: string } } })?.response?.data?.error ?? 'Gagal membuat penjualan hotspot'
      toast.error(message)
    },
  })
}

}

import { listMikhmonProfiles, deleteMikhmonProfile, type MikhmonProfileRow } from '@/api/mikhmon'

export function useMikhmonProfiles(routerId: string | null) {
  return useQuery({
    queryKey: ['mikhmon', routerId, 'profiles'],
    queryFn: () => listMikhmonProfiles(routerId!),
    enabled: !!routerId,
    staleTime: 60 * 1000,
  })

export function useDeleteMikhmonProfile() {

      toast.error(message)
    },

}
