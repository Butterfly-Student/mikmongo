import { z } from "zod"
import { ApiResponseSchema } from "./auth"

export const PPPSecretInfoSchema = z.object({

  id: z.string().uuid(),
  customer_id: z.string().uuid(),
  plan_id: z.string().uuid(),
  router_id: z.string().uuid(),
  username: z.string(),
  static_ip: z.string().nullable().optional(),
  gateway: z.string().nullable().optional(),
  status: z.enum(["pending", "active", "isolated", "suspended", "terminated"]),

  expiry_date: z.string().nullable().optional(),
  billing_day: z.number().nullable().optional(),
  auto_isolate: z.boolean(),
  grace_period_days: z.number().nullable().optional(),
  suspend_reason: z.string().nullable().optional(),

  created_at: z.string(),

export const UpdateSubscriptionFormSchema = CreateSubscriptionFormSchema.partial().omit({

**Step 2: Create `dashboard/src/api/subscriptions.ts`.**

All functions take `routerId` as first param (subscriptions are router-scoped):

```typescript
import { adminClient } from "@/lib/axios/admin-client"
import { SubscriptionListSchema, SubscriptionDetailSchema, SubscriptionMessageSchema } from "@/lib/schemas/subscription"

export async function fetchSubscriptions(routerId: string, params: { limit: number; offset: number }) {
  const { data } = await adminClient.get(`/routers/${routerId}/subscriptions`, { params })
  const parsed = SubscriptionListSchema.parse(data)
  return { data: parsed.data, meta: parsed.meta! }
}

export async function fetchSubscription(routerId: string, id: string) {
  const { data } = await adminClient.get(`/routers/${routerId}/subscriptions/${id}`)
  return SubscriptionDetailSchema.parse(data).data
}

export async function createSubscription(routerId: string, body: Record<string, unknown>) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions`, body)
  return SubscriptionDetailSchema.parse(data).data
}

export async function updateSubscription(routerId: string, id: string, body: Record<string, unknown>) {
  const { data } = await adminClient.put(`/routers/${routerId}/subscriptions/${id}`, body)
  return SubscriptionDetailSchema.parse(data).data
}

// Lifecycle actions
export async function activateSubscription(routerId: string, id: string) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions/${id}/activate`)
  return SubscriptionMessageSchema.parse(data).data
}

export async function isolateSubscription(routerId: string, id: string, reason?: string) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions/${id}/isolate`, reason ? { reason } : {})
  return SubscriptionMessageSchema.parse(data).data
}

export async function restoreSubscription(routerId: string, id: string) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions/${id}/restore`)
  return SubscriptionMessageSchema.parse(data).data
}

export async function suspendSubscription(routerId: string, id: string, reason?: string) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions/${id}/suspend`, reason ? { reason } : {})
  return SubscriptionMessageSchema.parse(data).data
}

export async function terminateSubscription(routerId: string, id: string) {
  const { data } = await adminClient.post(`/routers/${routerId}/subscriptions/${id}/terminate`)
  return SubscriptionMessageSchema.parse(data).data
}
```

export const subscriptionKeys = {
  all: (routerId: string) => ["subscriptions", routerId] as const,
  lists: (routerId: string) => [...subscriptionKeys.all(routerId), "list"] as const,
  list: (routerId: string, params: { limit: number; offset: number }) => [...subscriptionKeys.lists(routerId), params] as const,
  details: (routerId: string) => [...subscriptionKeys.all(routerId), "detail"] as const,

}
