```typescript
import { z } from "zod"
import { ApiResponseSchema } from "./auth"

export const RouterSchema = z.object({
  id: z.string().uuid(),
  name: z.string(),
  address: z.string(),
  area: z.string().nullable().optional(),
  api_port: z.number(),
  rest_port: z.number(),
  username: z.string(),
  use_ssl: z.boolean(),
  is_master: z.boolean(),
  is_active: z.boolean(),
  status: z.string(),
  last_seen_at: z.string().nullable().optional(),
  notes: z.string().nullable().optional(),
  created_at: z.string(),
  updated_at: z.string(),
})

export type Router = z.infer<typeof RouterSchema>

// Router list has NO meta -- uses response.OK, not response.WithMeta
export const RouterListSchema = ApiResponseSchema(z.array(RouterSchema))
export const RouterDetailSchema = ApiResponseSchema(RouterSchema)
// For select and single router responses that return Router or null
export const RouterNullableSchema = ApiResponseSchema(RouterSchema.nullable())
export const RouterMessageSchema = ApiResponseSchema(z.object({ message: z.string() }))

export const CreateRouterFormSchema = z.object({
  name: z.string().min(1, "Nama router wajib diisi"),
  address: z.string().min(1, "Alamat/IP wajib diisi"),
  area: z.string().optional(),
  api_port: z.coerce.number().int().min(1).max(65535).default(8728),
  rest_port: z.coerce.number().int().min(1).max(65535).default(443),
  username: z.string().min(1, "Username wajib diisi"),
  password: z.string().min(1, "Password wajib diisi"),
  use_ssl: z.boolean().default(false),
  is_master: z.boolean().default(false),
  notes: z.string().optional(),
})

export const UpdateRouterFormSchema = CreateRouterFormSchema.partial().omit({ password: true }).extend({
  password: z.string().optional(), // password optional on edit
})
```

```typescript
export async function fetchRouters(params: { limit: number; offset: number }) {
  const { data } = await adminClient.get("/routers", { params })
  const parsed = RouterListSchema.parse(data)
  return {
    data: parsed.data,
    meta: parsed.meta ?? { total: parsed.data.length, limit: params.limit, offset: params.offset },
  }
}
```

```typescript
import { create } from "zustand"
```
