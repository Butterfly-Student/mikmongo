**Step 2: Create `dashboard/src/api/customers.ts`.**

Follow the `reports.ts` pattern -- use `adminClient`, parse with Zod:

```typescript
import { adminClient } from "@/lib/axios/admin-client"
import {
  CustomerListSchema, CustomerDetailSchema, CustomerMutationSchema, CustomerMessageSchema,
  type Customer,
} from "@/lib/schemas/customer"

export async function fetchCustomers(params: { limit: number; offset: number }) {
  const { data } = await adminClient.get("/customers", { params })
  const parsed = CustomerListSchema.parse(data)
  return { data: parsed.data, meta: parsed.meta! }
}

export async function fetchCustomer(id: string) {
  const { data } = await adminClient.get(`/customers/${id}`)
  const parsed = CustomerDetailSchema.parse(data)
  return parsed.data
}

export async function createCustomer(body: { full_name: string; phone: string; email?: string; address?: string }) {
  const { data } = await adminClient.post("/customers", body)
  return CustomerMutationSchema.parse(data).data
}

export async function updateCustomer(id: string, body: Record<string, unknown>) {
  const { data } = await adminClient.put(`/customers/${id}`, body)
  const parsed = CustomerDetailSchema.parse(data)
  return parsed.data
}

export async function deleteCustomer(id: string) {
  const { data } = await adminClient.delete(`/customers/${id}`)
  return CustomerMessageSchema.parse(data).data
}

export async function activateCustomer(id: string) {
  const { data } = await adminClient.post(`/customers/${id}/activate-account`)
  return CustomerMessageSchema.parse(data).data
}

export async function deactivateCustomer(id: string) {
  const { data } = await adminClient.post(`/customers/${id}/deactivate-account`)
  return CustomerMessageSchema.parse(data).data
}
```

**Step 3: Create `dashboard/src/hooks/useCustomers.ts`.**

Follow the TanStack Query pattern with `keepPreviousData`:

```typescript
import { useQuery, useMutation, useQueryClient, keepPreviousData } from "@tanstack/react-query"
import { fetchCustomers, fetchCustomer, createCustomer, updateCustomer, deleteCustomer, activateCustomer, deactivateCustomer } from "@/api/customers"

export const customerKeys = {
  all: ["customers"] as const,
  lists: () => [...customerKeys.all, "list"] as const,
  list: (params: { limit: number; offset: number }) => [...customerKeys.lists(), params] as const,
  details: () => [...customerKeys.all, "detail"] as const,
  detail: (id: string) => [...customerKeys.details(), id] as const,
}

export function useCustomerList(limit: number, offset: number) {
  return useQuery({
    queryKey: customerKeys.list({ limit, offset }),
    queryFn: () => fetchCustomers({ limit, offset }),
    placeholderData: keepPreviousData,
    staleTime: 30_000,
  })
}

export function useCustomer(id: string) {
  return useQuery({
    queryKey: customerKeys.detail(id),
    queryFn: () => fetchCustomer(id),
    enabled: !!id,
  })
}

// Mutations: useCreateCustomer, useUpdateCustomer, useDeleteCustomer, useActivateCustomer, useDeactivateCustomer
// Each calls useQueryClient and invalidates customerKeys.all on success.
```
